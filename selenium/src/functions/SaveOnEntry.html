storeCookieByName entryNo entryNo
store css=tr.TableEditorRow:${entryNo}>td.TableEditorCell-Target tdCell

store Click here to start translating emptyDisplay
waitForElementPresent ${tdCell}
click css=tr.TableEditorRow:${entryNo}>td.TableEditorCell-Target

controlKeyDown
keyDown ${tdArea} \13

keyUp ${tdArea} \13
controlKeyUp

storeEval storedVars['saveString'] == "" isSavingEmpty
gotoIf ${isSavingEmpty} SavingEmpty

verifyText ${tdCell}>div ${emptyDisplay}
